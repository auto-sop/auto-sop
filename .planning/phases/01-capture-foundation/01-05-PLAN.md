```ts
import { statSync, renameSync, writeFileSync, appendFileSync, existsSync } from 'node:fs';
import { dirname } from 'node:path';
import { mkdirSync } from 'node:fs';

export const ERRORS_CAP_BYTES = 10 * 1024 * 1024; // 10 MB

export interface ErrorRecord {
  t: string;           // ISO timestamp
  kind: string;        // e.g. 'scrub_failed','rename_failed','disk_full','paused_skipped','zod_parse_failed','writer_uncaught'
  turn_id: string | null;
  err: string;         // .message if Error, else String(err)
}

export function logError(
  paths: { projectErrorsLog: string; globalErrorsLog: string },
  record: Omit<ErrorRecord, 't'>
): void {
  const line = JSON.stringify({ t: new Date().toISOString(), ...record }) + '\n';
  for (const target of [paths.projectErrorsLog, paths.globalErrorsLog]) {
    try {
      rotateIfNeeded(target);
      ensureFile(target);
      appendFileSync(target, line, { mode: 0o600 });
    } catch {
      // Silent — error logging must never itself throw.
    }
  }
}

function ensureFile(path: string): void {
  mkdirSync(dirname(path), { recursive: true, mode: 0o700 });
  if (!existsSync(path)) {
    writeFileSync(path, '', { mode: 0o600, flag: 'wx' });
  }
}

function rotateIfNeeded(path: string): void {
  if (!existsSync(path)) return;
  const { size } = statSync(path);
  if (size < ERRORS_CAP_BYTES) return;
  try {
    renameSync(path, path + '.1'); // overwrites any prior .1 — single backup policy
  } catch {
    // If rotation fails, continue — errors logging must not crash.
  }
}
```

The both-writes-best-effort pattern means if one path fails the other still gets written. The outer try/catch swallows errors because `logError` is called from inside `main`'s top-level catch — if it threw, we'd crash the writer process, which is acceptable (detached, doesn't affect CC) but sloppy.

Create `src/capture/writer/disk-budget.ts`:

```ts
import { existsSync, statSync, readdirSync, writeFileSync, unlinkSync } from 'node:fs';
import { join } from 'node:path';

export const DEFAULT_CAP_BYTES = 2 * 1024 * 1024 * 1024; // 2 GB
export const PAUSE_RATIO = 0.5;                           // 50% of cap

export function isPaused(pausedFlagPath: string): boolean {
  return existsSync(pausedFlagPath);
}

export function computeUsedBytes(dir: string): number {
  // Recursive du; swallow ENOENT
  if (!existsSync(dir)) return 0;
  let total = 0;
  const stack: string[] = [dir];
  while (stack.length) {
    const d = stack.pop()!;
    let entries: string[];
    try { entries = readdirSync(d); } catch { continue; }
    for (const name of entries) {
      const p = join(d, name);
      try {
        const st = statSync(p);
        if (st.isDirectory()) stack.push(p);
        else total += st.size;
      } catch { /* race: file vanished between readdir and stat */ }
    }
  }
  return total;
}

export function enforceDiskBudget(
  capturesDir: string,
  pausedFlagPath: string,
  capBytes: number = DEFAULT_CAP_BYTES
): { used: number; paused: boolean; justPaused: boolean } {
  const used = computeUsedBytes(capturesDir);
  const threshold = Math.floor(capBytes * PAUSE_RATIO);
  const alreadyPaused = existsSync(pausedFlagPath);
  if (used >= threshold && !alreadyPaused) {
    try {
      writeFileSync(pausedFlagPath, JSON.stringify({ at: new Date().toISOString(), used, cap: capBytes, threshold }), { mode: 0o600 });
    } catch { /* if we can't write flag, next call will try again */ }
    return { used, paused: true, justPaused: true };
  }
  return { used, paused: alreadyPaused, justPaused: false };
}
```

Note: `computeUsedBytes` walks the captures dir — potentially expensive on large stores. Call it ONLY on UserPromptSubmit (once per turn start), not on every PreToolUse. Cache is out of scope.
