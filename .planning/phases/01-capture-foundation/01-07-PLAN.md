/**
 * For an event carrying agent_id, look up (or create-intent for) the subagent's turn.
 * Uses a per-session, per-agent-id marker: current-turn-<session_id>-<agent_id>.json
 */
export function resolveSubagentTurn(
  stateDir: string,
  sessionId: string,
  agentId: string
): { turnDir: string; turnId: string } | null { /* read current-turn-<sess>-<agent>.json */ }

export function setSubagentCurrentTurn(
  stateDir: string,
  sessionId: string,
  agentId: string,
  data: { turnDir: string; turnId: string }
): void { /* write current-turn-<sess>-<agent>.json */ }

export function clearSubagentCurrentTurn(stateDir: string, sessionId: string, agentId: string): void { /* unlink */ }

/**
 * On SubagentStop: the child's dir is already written; we now:
 * 1. Finalize the child meta + rename its .pending dir
 * 2. Append child's turn_id to parent's meta.children_turn_ids (read-modify-write)
 */
export function linkChildToParent(
  childMeta: TurnMeta,
  parentTurnDir: string | null
): void {
  if (!parentTurnDir) return; // orphan subagent — no parent found, log kind:'orphan_subagent'
  const parent = readMeta(parentTurnDir);
  if (!parent) return;
  const updated = {
    children_turn_ids: Array.from(new Set([...(parent.children_turn_ids ?? []), childMeta.turn_id])),
  };
  updateMeta(parentTurnDir, updated);
}
```

import { readdirSync, statSync, renameSync, unlinkSync, existsSync, mkdirSync } from 'node:fs';
import { join } from 'node:path';
import { readMeta, finalizeMeta } from './meta.js';
import { finalizeTurnDir } from './turn-dir.js';

export const STALE_TIMEOUT_MS = 30_000;        // 30 seconds → finalize as timeout
export const STALE_YARIM_KALAN_MS = 30 * 60_000; // 30 minutes → quarantine
export const TMP_MAX_AGE_MS = 60 * 60_000;      // 1 hour
export const MAX_TMP_SWEEP_PER_PASS = 50;

export interface SweepResult {
  finalized: number;
  quarantined: number;
  errors: number;
}

export function sweepOrphanedTurns(
  capturesDir: string,
  yarimKalanDir: string,
  now: number = Date.now()
): SweepResult { /* ... */ }

export function sweepOrphanTmpPayloads(
  tmpDir: string,
  now: number = Date.now()
): { deleted: number } { /* ... */ }
```

`sweepOrphanedTurns` behavior:
- `readdirSync(capturesDir)` → filter names ending in `.pending`.
- For each pending dir, compute `lastMtime = max(mtime of dir itself, mtime of files inside)`. Walk one level deep — don't recurse into large-outputs/.
- If `now - lastMtime > STALE_YARIM_KALAN_MS`: move the dir to `<yarimKalanDir>/<same-basename>` (mkdir yarimKalanDir if missing, mode 0o700; then `renameSync`). Increment `quarantined`.
- Else if `now - lastMtime > STALE_TIMEOUT_MS`: `finalizeMeta(dir, 'timeout')`, then `finalizeTurnDir(dir)` (drop .pending). Increment `finalized`.
- Else: skip (still active).
- Any failure → catch, increment `errors`, continue.

`sweepOrphanTmpPayloads` behavior:
- `readdirSync(tmpDir)` (if exists).
- For each file, if `now - stat.mtimeMs > TMP_MAX_AGE_MS`: `unlinkSync`. Stop after `MAX_TMP_SWEEP_PER_PASS` deletions to bound latency.
- Return count.

import { registerPreStartHook } from './pre-start-hooks.js';
registerPreStartHook((event, ctx) => {

    sweepOrphanedTurns(ctx.paths.projectCaptureDir, ctx.paths.projectYarimKalan);

    (ctx as any).errorWriter?.('orphan_sweep_failed', null, err);

});
