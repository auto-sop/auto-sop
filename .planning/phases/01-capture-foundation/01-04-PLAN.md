```ts
import { appendFileSync, writeFileSync } from 'node:fs';
import { join } from 'node:path';
import type { Scrubber } from '../../scrubber/index.js';

// W4: NO lockfile and NO PIPE_BUF threshold for turn-local JSONL files.
// Single-writer-per-turn (Pattern 4) guarantees no concurrent writes; plain O_APPEND is safe.
// Lockfile is only needed for the GLOBAL index.jsonl (plan 01-06), which multiple projects may append to concurrently.

export interface PreToolLine {
  event: 'pre';
  tool_use_id: string;
  tool: string;
  input: unknown;
  t: string; // ISO timestamp
}

export interface PostToolLine {
  event: 'post';
  tool_use_id: string;
  output?: unknown;       // present only when inline (under threshold)
  output_ref?: string;    // relative path to large-outputs/<id>.out.txt.gz
  input_ref?: string;     // relative path to large-outputs/<id>.in.txt.gz (set if input was offloaded)
  bytes?: number;         // total original bytes when ref'd
  duration_ms?: number;
  success: boolean;
  t: string;
}

export function appendPreToolLine(turnDir: string, line: PreToolLine, scrubber: Scrubber): {hitCount: number} { /* ... */ }
export function appendPostToolLine(turnDir: string, line: PostToolLine, scrubber: Scrubber): {hitCount: number} { /* ... */ }
```
(Both are synchronous now — no lockfile async ceremony.)

Implementation notes:
- The line is scrubbed AT THE STRING LEVEL: serialize the object to JSON first, scrub the resulting string, verify it's still valid JSON (parse+re-stringify to normalize) → if scrubber broke JSON validity (unlikely but possible), fall back to `{event:'pre',tool_use_id,tool,input:'[REDACTION_BROKE_JSON]',t}`. Track hit count.
- Append with a terminating `\n`.
- **No size gate, no lockfile:** `appendFileSync(jsonlPath, line+'\n')` unconditionally. Under Pattern 4 (single-writer-per-turn), this process is the only writer for this file right now. The POSIX O_APPEND semantics mean even if something weird happened, writes are ordered — but the contract is "one writer per turn, period."
- Always chmod-on-create: if file doesn't exist, `writeFileSync(jsonlPath, '', {mode: 0o600, flag: 'wx'})` first (swallow EEXIST). Guarantees 0600.
- The `turnDir` arg is the `.pending` directory from session state; `tool-calls-route.ts` resolves current-turn before calling.
- Timestamp: `new Date().toISOString()`.

```ts
import { createWriteStream, mkdirSync } from 'node:fs';
import { createGzip } from 'node:zlib';
import { join } from 'node:path';
import { pipeline } from 'node:stream/promises';
import { Readable } from 'node:stream';

export const LARGE_OUTPUT_THRESHOLD = 262144; // 256 KB

export interface OffloadResult {
  offloaded: boolean;
  ref?: string;   // relative path inside turn dir, e.g. 'large-outputs/abc.in.txt.gz'
  bytes?: number; // original byte count
}

export async function maybeOffloadLarge(
  turnDir: string,
  toolUseId: string,
  kind: 'in' | 'out',
  content: string | Buffer,
  threshold: number = LARGE_OUTPUT_THRESHOLD,
): Promise<OffloadResult> { /* ... */ }
```

Implementation:
- Convert `content` to Buffer if string (`Buffer.from(content, 'utf8')`).
- Measure `bytes = buf.length`. If `bytes < threshold`, return `{offloaded: false}`.
- `mkdirSync(join(turnDir, 'large-outputs'), {recursive:true, mode:0o700})`.
- `destPath = join(turnDir, 'large-outputs', `${toolUseId}.${kind}.txt.gz`)`.
- Stream the buffer through gzip to `createWriteStream(destPath, {mode:0o600})` via `stream.pipeline` — use `Readable.from(buf)`.
- Return `{offloaded: true, ref: 'large-outputs/' + basename, bytes}`.

**Scrubbing the large payload:** IMPORTANT — large content must ALSO be scrubbed. Accept a `scrubber` parameter, scrub the string form before byte-counting. Record hit count, return it in OffloadResult. Signature becomes:

```ts
export async function maybeOffloadLarge(
  turnDir: string, toolUseId: string, kind: 'in'|'out',
  rawContent: string | Buffer, scrubber: Scrubber, threshold?: number
): Promise<OffloadResult & {hitCount: number}>
```
