```ts
import { appendFileSync, mkdirSync, writeFileSync, existsSync } from 'node:fs';
import { dirname } from 'node:path';
import type { TurnMeta } from '../types.js';
import type { CapturePaths } from '../paths.js';

export interface GlobalIndexLine {
  turn_id: string;
  session_id: string;
  project_id: string;
  project_path: string;             // absolute project root
  project_turn_dir: string;         // absolute path to the finalized (non-.pending) turn dir
  agent: string;
  parent_turn_id: string | null;
  finalization_reason: string;
  t: string;                        // finalized_at
}

export function resolveGlobalTargetDir(
  paths: CapturePaths,
  projectRoot: string,
  detectDevArmyAgent: (root: string) => string | null
): string {
  const devArmyAgent = detectDevArmyAgent(projectRoot);
  return devArmyAgent ? paths.devArmyGlobalDir(devArmyAgent) : paths.globalProjectDir;
}

export function appendGlobalIndexLine(
  paths: CapturePaths,
  projectRoot: string,
  meta: TurnMeta,
  turnDirAbs: string,
  detectDevArmyAgent: (root: string) => string | null
): void {
  const targetDir = resolveGlobalTargetDir(paths, projectRoot, detectDevArmyAgent);
  const indexPath = `${targetDir}/index.jsonl`;
  mkdirSync(targetDir, { recursive: true, mode: 0o700 });
  if (!existsSync(indexPath)) writeFileSync(indexPath, '', { mode: 0o600, flag: 'wx' });

  const line: GlobalIndexLine = {
    turn_id: meta.turn_id,
    session_id: meta.session_id,
    project_id: meta.project_id,
    project_path: projectRoot,
    project_turn_dir: turnDirAbs,
    agent: meta.agent,
    parent_turn_id: meta.parent_turn_id,
    finalization_reason: meta.finalization_reason ?? 'unknown',
    t: meta.finalized_at ?? new Date().toISOString(),
  };
  appendFileSync(indexPath, JSON.stringify(line) + '\n', { mode: 0o600 });
}
```

Extend `src/capture/writer/global-mirror.ts` with:

```ts
export function migrateGlobalDirOnMove(
  oldGlobalDir: string,
  newGlobalDir: string,
  newProjectPath: string
): { moved: boolean; linesRewritten: number } { /* ... */ }
```

Behavior:
- If `oldGlobalDir === newGlobalDir`, return `{moved: false, linesRewritten: 0}`.
- If `!existsSync(oldGlobalDir)`, return `{moved: false, linesRewritten: 0}` (nothing to migrate).
- If `existsSync(newGlobalDir)`: collision — append a `migration.log` line to `newGlobalDir` saying "collision with <old>", DO NOT move, return `{moved: false, ...}`. We don't want to clobber existing data.
- Otherwise: `renameSync(oldGlobalDir, newGlobalDir)` — POSIX-atomic if on same filesystem (both under `~/.claude/sop/`, so yes).
- Read the new index.jsonl, parse each line as JSON, replace `project_path` with `newProjectPath`, rewrite `project_turn_dir` by replacing the old project prefix with the new prefix (simple string replace). Write to `index.jsonl.tmp`, `renameSync` atomically.
- Append a line to `<newGlobalDir>/migration.log`: `<iso> moved from <oldGlobalDir> to <newGlobalDir>, rewrote N lines`.
- Return `{moved: true, linesRewritten: N}`.

**W3 resolution:** `detectMove` is already exported from `src/path-resolver/index.ts` — wire it directly. No "deferred with TODO" escape clause.

Add to `src/capture/writer/routes/global-mirror-hook.ts` a pre-start hook registration (using 01-03's `registerPreStartHook`) that runs on every event:
```ts
import { registerPreStartHook } from './pre-start-hooks.js';
import { detectMove } from '../../../path-resolver/index.js';
import { migrateGlobalDirOnMove } from '../global-mirror.js';

registerPreStartHook((event, ctx) => {
  const move = detectMove(ctx.projectRoot);
  if (move?.moved) {
    try {
      migrateGlobalDirOnMove(move.oldGlobalDir, move.newGlobalDir, ctx.projectRoot);
      (ctx as any).errorWriter?.('project_moved', null, `${move.oldGlobalDir} -> ${move.newGlobalDir}`);
    } catch (err) {
      (ctx as any).errorWriter?.('project_move_migration_failed', null, err);
    }
  }
  return { abort: false };
});
```
(If `detectMove`'s actual return shape differs from `{moved, oldGlobalDir, newGlobalDir}`, compute the two global dirs here from the old/new identities using `getCapturePaths`.) **Wire directly — no TODO deferral.** main.ts still not touched.
