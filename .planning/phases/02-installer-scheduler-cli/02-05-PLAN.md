     ```ts
     import { promises as fs } from 'node:fs';
     import path from 'node:path';

     export async function copyPluginBundle(srcDir: string, dstDir: string): Promise<void> {
       // Validate src exists and is a directory
       const stat = await fs.stat(srcDir);
       if (!stat.isDirectory()) throw new Error(`plugin bundle source is not a directory: ${srcDir}`);
       // Destructive copy: remove existing dst, then fs.cp recursive
       await fs.rm(dstDir, { recursive: true, force: true });
       await fs.mkdir(path.dirname(dstDir), { recursive: true });
       await fs.cp(srcDir, dstDir, { recursive: true, force: true });
     }
     ```
     Node ≥18.17 has `fs.cp` recursive — do NOT add fs-extra.

  1. Create `src/cli/verbs/install.ts`:
     ```ts
     import type { Command } from 'commander';
     import path from 'node:path';
     import os from 'node:os';
     import { fileURLToPath } from 'node:url';
     import { readFileSync } from 'node:fs';
     import { runInstall } from '../../installer/orchestrator.js';
     import { emit } from '../output/json.js';
     import { renderTable, warn } from '../output/human.js';
     import pc from 'picocolors';

     export function registerInstallVerb(program: Command): void {
       program
         .command('install')
         .description('install claude-sop into the current project')
         .option('--license <key>', 'license key (non-interactive; skips prompt)')
         .option('--project <path>', 'project root', process.cwd())
         .action(async (opts, cmd) => {
           const jsonMode = cmd.parent?.opts().json ?? false;
           // Resolve package root (two or three levels up from this file in dist tree)
           const here = path.dirname(fileURLToPath(import.meta.url));
           const packageRoot = findPackageRoot(here);
           const packageVersion = JSON.parse(readFileSync(path.join(packageRoot, 'package.json'), 'utf8')).version;
           const pluginBundleSrc = path.join(packageRoot, 'dist', 'plugin');
           const homeDir = os.homedir();
           const marketplaceDir = path.join(homeDir, '.claude-sop', 'marketplace', 'claude-sop');
           const result = await runInstall({
             projectRoot: path.resolve(opts.project),
             homeDir,
             licenseKey: opts.license,
             pluginBundleSrc,
             packageVersion,
             nodeBin: process.execPath,
             shimAbsPath: path.join(marketplaceDir, 'shim.cjs'),
             learnerAbsPath: path.join(marketplaceDir, 'learner.cjs'),
           });
           if (jsonMode) {
             emit({ ok: true, verb: 'install', ...result });
           } else {
             process.stdout.write(pc.green(`✓ claude-sop ${result.verdict} install complete\n`));
             process.stdout.write(renderTable([
               ['version', result.installedVersion],
               ['scheduler', result.scheduler],
               ['managed section', result.managedSection],
               ['.gitignore', result.gitignore],
             ]) + '\n');
             for (const w of result.warnings) warn(w);
           }
         });
     }

     function findPackageRoot(startDir: string): string {
       let dir = startDir;
       for (let i = 0; i < 6; i++) {
         try {
           const pkg = path.join(dir, 'package.json');
           readFileSync(pkg, 'utf8');
           return dir;
         } catch {
           dir = path.dirname(dir);
         }
       }
       throw new Error('cannot locate package root from ' + startDir);
     }
     ```
