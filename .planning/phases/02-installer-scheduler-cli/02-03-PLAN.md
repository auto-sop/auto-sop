     ```ts
     import { execa } from 'execa';
     export async function systemdUserAvailable(): Promise<boolean> {
       try {
         const r = await execa('systemctl', ['--user', 'is-system-running'], {
           reject: false,
           timeout: 2000,
         });
         // running, degraded, starting, initializing, maintenance all mean bus is up
         if (r.exitCode === 0) return true;
         return /running|degraded|starting|initializing|maintenance/.test(r.stdout || '');
       } catch {
         return false;
       }
     }
     ```

     ```ts
     import { writeFileAtomic } from '../atomic/write.js';
     import { promises as fs } from 'node:fs';

     export interface TickScriptOpts {
       homeDir: string;
       nodeBin: string;       // process.execPath at install time
       learnerJs: string;     // absolute path to learner entry
       errorsLog: string;     // absolute path
     }

     export function renderTickScript(opts: TickScriptOpts): string {
       // POSIX sh; quoted variable expansions; absolute paths baked in.
       return [
         '#!/bin/sh',
         '# claude-sop hourly tick wrapper — generated by `npx claude-sop install`',
         '# Do not edit; regenerated on every re-install.',
         '',
         'set -eu',
         '',
         '# Minimal, deterministic env',
         'export PATH="/usr/local/bin:/usr/bin:/bin:${PATH:-}"',
         `export HOME=${shellQuote(opts.homeDir)}`,
         'export CLAUDE_SOP_LEARNER=1',
         '',
         '# Absolute paths baked in at install time',
         `NODE_BIN=${shellQuote(opts.nodeBin)}`,
         `LEARNER_JS=${shellQuote(opts.learnerJs)}`,
         `ERRORS_LOG=${shellQuote(opts.errorsLog)}`,
         '',
         'mkdir -p "$(dirname "$ERRORS_LOG")"',
         '',
         '# Locking is done INSIDE the learner via proper-lockfile (macOS has no flock(1)).',
         'exec "$NODE_BIN" "$LEARNER_JS" 2>>"$ERRORS_LOG"',
         '',
       ].join('\n');
     }

     function shellQuote(s: string): string {
       // POSIX-safe single-quote wrapping: close, escape, reopen.
       return `'${s.replace(/'/g, `'\\''`)}'`;
     }

     export async function writeTickScript(path: string, opts: TickScriptOpts): Promise<void> {
       await writeFileAtomic(path, renderTickScript(opts));
       await fs.chmod(path, 0o755);
     }
     ```

     ```ts
     import { systemdUserAvailable } from './detect.js';
     import { macosLaunchd } from './macos-launchd.js';
     import { linuxSystemd } from './linux-systemd.js';
     import { linuxCron } from './linux-cron.js';
     import type { SchedulerBackend } from './types.js';

     export type { SchedulerBackend, SchedulerStatus, SchedulerInstallOpts } from './types.js';
     export { systemdUserAvailable, macosLaunchd, linuxSystemd, linuxCron };

     export async function pickBackend(platform: NodeJS.Platform = process.platform): Promise<{ backend: SchedulerBackend; fallbackWarning?: string }> {
       if (platform === 'darwin') return { backend: macosLaunchd };
       if (platform === 'linux') {
         if (await systemdUserAvailable()) return { backend: linuxSystemd };
         return {
           backend: linuxCron,
           fallbackWarning:
             'systemd --user is unavailable on this system. claude-sop installed an hourly cron entry as a fallback. Reboot-persistence depends on your distribution\'s cron daemon configuration. For best reliability, enable systemd --user or use a system with lingering support.',
         };
       }
       throw new Error(`unsupported platform: ${platform}`);
     }
     ```
