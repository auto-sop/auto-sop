     ```ts
     import { promises as fs } from 'node:fs';
     import path from 'node:path';
     import os from 'node:os';
     import { nanoid } from 'nanoid';
     import type { SchedulerBackend, SchedulerStatus, SchedulerInstallOpts } from '../../src/scheduler/types.js';

     export async function makeTempHome(): Promise<{ homeDir: string; projectRoot: string; cleanup: () => Promise<void> }> {
       const root = await fs.mkdtemp(path.join(os.tmpdir(), 'claude-sop-e2e-'));
       const homeDir = path.join(root, 'home');
       const projectRoot = path.join(root, 'project');
       await fs.mkdir(homeDir, { recursive: true });
       await fs.mkdir(projectRoot, { recursive: true });
       return {
         homeDir,
         projectRoot,
         cleanup: async () => { await fs.rm(root, { recursive: true, force: true }); },
       };
     }

     export async function seedPluginBundleFixture(destPkgRoot: string): Promise<string> {
       // Copy fixtures/plugin-bundle to <destPkgRoot>/dist/plugin so the orchestrator sees it
       const src = path.join(__dirname, 'fixtures', 'plugin-bundle');
       const dst = path.join(destPkgRoot, 'dist', 'plugin');
       await fs.mkdir(path.dirname(dst), { recursive: true });
       await fs.cp(src, dst, { recursive: true, force: true });
       return dst;
     }

     export function stubSchedulerBackend(overrides: Partial<SchedulerBackend> = {}): SchedulerBackend & { calls: { install: SchedulerInstallOpts[]; uninstall: number; status: number } } {
       const calls = { install: [] as SchedulerInstallOpts[], uninstall: 0, status: 0 };
       return {
         name: 'launchd' as const,
         async install(opts) { calls.install.push(opts); },
         async uninstall() { calls.uninstall++; return { warnings: [] }; },
         async status(): Promise<SchedulerStatus> {
           calls.status++;
           return { backend: 'launchd', installed: calls.install.length > 0, lastTickAt: null, lastExitCode: null, details: {} };
         },
         calls,
         ...overrides,
       } as any;
     }
     ```

  ```ts
  import { describe, it, expect, beforeAll, afterAll, beforeEach, afterEach } from 'vitest';
  import { promises as fs } from 'node:fs';
  import path from 'node:path';
  import os from 'node:os';
  import { parse } from 'jsonc-parser';
  import { createHash } from 'node:crypto';
  import { makeTempHome, seedPluginBundleFixture, stubSchedulerBackend } from './helpers.js';
  import { runInstall } from '../../src/installer/orchestrator.js';
  import { runUninstall } from '../../src/installer/uninstall-orchestrator.js';
  import { collectStatus } from '../../src/status/collector.js';
  import { HOOK_EVENTS, CLAUDE_SOP_HOOK_ID } from '../../src/installer/hook-entries.js';
  import { MANAGED_BEGIN, MANAGED_END } from '../../src/installer/managed-section.js';

  describe('Phase 2 e2e — install → status → pause → resume → uninstall', () => {
    let tmp: Awaited<ReturnType<typeof makeTempHome>>;
    let pluginBundleSrc: string;
    const packageVersion = '0.1.0';

    beforeEach(async () => {
      tmp = await makeTempHome();
      // Pretend our package root is under tmp so we can seed dist/plugin
      const fakePkgRoot = path.join(tmp.homeDir, '..', 'pkg');
      await fs.mkdir(fakePkgRoot, { recursive: true });
      pluginBundleSrc = await seedPluginBundleFixture(fakePkgRoot);
    });

    afterEach(async () => { await tmp.cleanup(); });

    async function hashFile(p: string): Promise<string> {
      const buf = await fs.readFile(p);
      return createHash('sha256').update(buf).digest('hex');
    }

    function makeInstallOpts(scheduler = stubSchedulerBackend()) {
      const homeDir = tmp.homeDir;
      const marketplaceDir = path.join(homeDir, '.claude-sop', 'marketplace', 'claude-sop');
      return {
        opts: {
          projectRoot: tmp.projectRoot,
          homeDir,
          licenseKey: '123',
          pluginBundleSrc,
          packageVersion,
          nodeBin: process.execPath,
          shimAbsPath: path.join(marketplaceDir, 'shim.cjs'),
          learnerAbsPath: path.join(marketplaceDir, 'learner.cjs'),
          promptLicense: async () => '123',
          schedulerBackend: scheduler,
          now: 1_700_000_000_000,
        },
        scheduler,
        marketplaceDir,
      };
    }

    it('INST-01: install wires hooks, creates secrets.enc with trial, prompts license', async () => {
      const { opts, marketplaceDir, scheduler } = makeInstallOpts();
      const result = await runInstall(opts);
      expect(result.verdict).toBe('fresh');
      // Plugin bundle copied
      expect((await fs.stat(path.join(marketplaceDir, 'shim.cjs'))).isFile()).toBe(true);
      // scheduler stub called
      expect(scheduler.calls.install).toHaveLength(1);
      // secrets.enc exists
      expect((await fs.stat(path.join(tmp.homeDir, '.claude-sop', 'secrets.enc'))).isFile()).toBe(true);
    });

    it('INST-02: install is idempotent (re-run produces byte-identical settings.json)', async () => {
      const { opts } = makeInstallOpts();
      await runInstall(opts);
      const projectSettings = path.join(tmp.projectRoot, '.claude', 'settings.json');
      const firstHash = await hashFile(projectSettings);
      await runInstall({ ...opts, schedulerBackend: stubSchedulerBackend() });
      const secondHash = await hashFile(projectSettings);
      expect(secondHash).toBe(firstHash);
    });

    it('INST-03: install preserves pre-existing user hooks in settings.json', async () => {
      // Seed project settings.json with a user hook
      const settingsDir = path.join(tmp.projectRoot, '.claude');
      await fs.mkdir(settingsDir, { recursive: true });
      const userHook = {
        hooks: {
          UserPromptSubmit: [{ hooks: [{ type: 'command', command: '/usr/local/bin/my-hook', timeout: 5 }] }],
        },
      };
      await fs.writeFile(path.join(settingsDir, 'settings.json'), JSON.stringify(userHook, null, 2));
      const { opts } = makeInstallOpts();
      await runInstall(opts);
      const after = JSON.parse(await fs.readFile(path.join(settingsDir, 'settings.json'), 'utf8'));
      expect(after.hooks.UserPromptSubmit).toHaveLength(2);
      expect(after.hooks.UserPromptSubmit[0].hooks[0].command).toBe('/usr/local/bin/my-hook'); // user first
      expect(after.hooks.UserPromptSubmit[1].hooks[0].id).toBe(CLAUDE_SOP_HOOK_ID);           // ours last
    });

    it('INST-04: install appends .claude-sop/ to project .gitignore', async () => {
      const { opts } = makeInstallOpts();
      await runInstall(opts);
      const gitignore = await fs.readFile(path.join(tmp.projectRoot, '.gitignore'), 'utf8');
      expect(gitignore).toContain('.claude-sop/');
    });

    it('INST-05: CLAUDE.md has empty managed-section markers', async () => {
      const { opts } = makeInstallOpts();
      await runInstall(opts);
      const md = await fs.readFile(path.join(tmp.projectRoot, 'CLAUDE.md'), 'utf8');
      expect(md).toContain(MANAGED_BEGIN);
      expect(md).toContain(MANAGED_END);
    });

    it('SCHED-01/04: scheduler install called with absolute tick.sh path', async () => {
      const { opts, scheduler } = makeInstallOpts();
      await runInstall(opts);
      const call = scheduler.calls.install[0];
      expect(call.tickScriptPath.startsWith('/')).toBe(true);
      expect(call.tickScriptPath.endsWith('/bin/tick.sh')).toBe(true);
      expect(call.intervalSec).toBe(3600);
    });

    it('SCHED-03: tick.sh file exists and is executable, and locking is delegated to Node (proper-lockfile) not flock', async () => {
      const { opts } = makeInstallOpts();
      await runInstall(opts);
      const tick = path.join(tmp.homeDir, '.claude-sop', 'bin', 'tick.sh');
      const st = await fs.stat(tick);
      expect(st.isFile()).toBe(true);
      expect((st.mode & 0o111) !== 0).toBe(true); // executable bit set
      const content = await fs.readFile(tick, 'utf8');
      expect(content).not.toContain('flock');           // H4: no flock(1)
      expect(content).toContain('CLAUDE_SOP_LEARNER=1'); // H2
      expect(content.split('\n')[0]).toBe('#!/bin/sh');
    });

    it('LIC-01/02: secrets.enc contains schema v1 with trial.started_at and license.kind=dev for key 123', async () => {
      const { opts } = makeInstallOpts();
      await runInstall(opts);
      const { readSecrets } = await import('../../src/license/storage.js');
      const payload = await readSecrets(path.join(tmp.homeDir, '.claude-sop', 'secrets.enc'));
      expect(payload!.schema_version).toBe(1);
      expect(payload!.license.kind).toBe('dev');
      expect(payload!.trial.started_at).toBe(1_700_000_000_000);
      expect(payload!.trial.duration_days).toBe(14);
    });

    it('LIC-02: re-install preserves trial.started_at', async () => {
      const { opts } = makeInstallOpts();
      await runInstall(opts);
      await runInstall({ ...opts, now: 1_700_000_999_999, schedulerBackend: stubSchedulerBackend() });
      const { readSecrets } = await import('../../src/license/storage.js');
      const payload = await readSecrets(path.join(tmp.homeDir, '.claude-sop', 'secrets.enc'));
      expect(payload!.trial.started_at).toBe(1_700_000_000_000); // unchanged
    });

    it('CLI-01: status returns all I3 fields after install', async () => {
      const { opts, scheduler } = makeInstallOpts();
      await runInstall(opts);
      const report = await collectStatus({
        projectRoot: tmp.projectRoot,
        homeDir: tmp.homeDir,
        projectHash12: 'abcdef012345',
        projectSlug: 'test-project',
        schedulerBackend: scheduler,
      });
      expect(report.installedVersion).toBe(packageVersion);
      expect(report.hooks.wiringState).toBe('present');
      expect(report.hooks.eventsCovered).toHaveLength(HOOK_EVENTS.length);
      expect(report.license.status).toBe('dev-key');
      expect(report.paused).toBe(false);
    });

    it('CLI-05: pause + resume toggle paused.flag and status reflects it', async () => {
      const { opts, scheduler } = makeInstallOpts();
      await runInstall(opts);
      const flagPath = path.join(tmp.projectRoot, '.claude-sop', 'paused.flag');
      await fs.writeFile(flagPath, JSON.stringify({ paused_at: Date.now() }));
      let report = await collectStatus({
        projectRoot: tmp.projectRoot,
        homeDir: tmp.homeDir,
        projectHash12: 'abcdef012345',
        projectSlug: 'test-project',
        schedulerBackend: scheduler,
      });
      expect(report.paused).toBe(true);
      await fs.rm(flagPath);
      report = await collectStatus({
        projectRoot: tmp.projectRoot,
        homeDir: tmp.homeDir,
        projectHash12: 'abcdef012345',
        projectSlug: 'test-project',
        schedulerBackend: scheduler,
      });
      expect(report.paused).toBe(false);
    });

    it('INST-06: default uninstall removes hooks, scheduler, managed section, secrets.enc but preserves captures', async () => {
      const { opts, scheduler } = makeInstallOpts();
      await runInstall(opts);
      // Seed a capture file
      const capturesDir = path.join(tmp.projectRoot, '.claude-sop', 'captures');
      await fs.mkdir(capturesDir, { recursive: true });
      await fs.writeFile(path.join(capturesDir, 'turn-001.json'), '{"id":"001"}');

      const r = await runUninstall({
        projectRoot: tmp.projectRoot,
        homeDir: tmp.homeDir,
        purge: false,
        projectHash12: 'abcdef012345',
        schedulerBackend: scheduler,
      });
      expect(r.warnings).toHaveLength(0);

      // Hooks removed from settings.json
      const settings = parse(await fs.readFile(path.join(tmp.projectRoot, '.claude', 'settings.json'), 'utf8'), [], { allowTrailingComma: true });
      for (const ev of HOOK_EVENTS) {
        const arr = settings?.hooks?.[ev] ?? [];
        for (const entry of arr) {
          for (const h of entry.hooks ?? []) {
            expect(h.id).not.toBe(CLAUDE_SOP_HOOK_ID);
          }
        }
      }
      // Managed section stripped
      const md = await fs.readFile(path.join(tmp.projectRoot, 'CLAUDE.md'), 'utf8');
      expect(md).not.toContain(MANAGED_BEGIN);
      // secrets.enc removed
      await expect(fs.stat(path.join(tmp.homeDir, '.claude-sop', 'secrets.enc'))).rejects.toThrow();
      // tick.sh removed
      await expect(fs.stat(path.join(tmp.homeDir, '.claude-sop', 'bin', 'tick.sh'))).rejects.toThrow();
      // version.txt removed
      await expect(fs.stat(path.join(tmp.homeDir, '.claude-sop', 'version.txt'))).rejects.toThrow();
      // Captures preserved
      expect((await fs.readFile(path.join(capturesDir, 'turn-001.json'), 'utf8')).length).toBeGreaterThan(0);
      // Scheduler backend uninstall called
      expect(scheduler.calls.uninstall).toBe(1);
    });

    it('INST-06 + PRIV-06: --purge also wipes captures', async () => {
      const { opts, scheduler } = makeInstallOpts();
      await runInstall(opts);
      const capturesDir = path.join(tmp.projectRoot, '.claude-sop', 'captures');
      await fs.mkdir(capturesDir, { recursive: true });
      await fs.writeFile(path.join(capturesDir, 'turn-001.json'), '{}');

      await runUninstall({
        projectRoot: tmp.projectRoot,
        homeDir: tmp.homeDir,
        purge: true,
        projectHash12: 'abcdef012345',
        schedulerBackend: scheduler,
      });
      await expect(fs.stat(capturesDir)).rejects.toThrow();
    });

    it('G3: downgrade is refused', async () => {
      const { opts } = makeInstallOpts();
      await runInstall({ ...opts, packageVersion: '9.9.9' });
      const { PreconditionError } = await import('../../src/cli/errors.js');
      await expect(runInstall({ ...opts, packageVersion: '0.0.1', schedulerBackend: stubSchedulerBackend() }))
        .rejects.toBeInstanceOf(PreconditionError);
    });
  });
  ```
