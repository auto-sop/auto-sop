  5. Create `src/cli/main.ts` — commander root, FROZEN after this plan:
     ```ts
     import { Command, CommanderError } from 'commander';
     import pc from 'picocolors';
     import { readFileSync } from 'node:fs';
     import { fileURLToPath } from 'node:url';
     import { dirname, join } from 'node:path';
     import { ExitCode } from './exit-codes.js';
     import { PreconditionError } from './errors.js';
     import { registerVerbs } from './verbs/index.js';
     import { emitError } from './output/json.js';

     function readPkgVersion(): string {
       // Walk up from this file to find package.json
       const here = dirname(fileURLToPath(import.meta.url));
       for (const p of [join(here, '..', '..', 'package.json'), join(here, '..', '..', '..', 'package.json')]) {
         try { return JSON.parse(readFileSync(p, 'utf8')).version; } catch { /* try next */ }
       }
       return '0.0.0';
     }

     export async function runCli(argv: string[]): Promise<number> {
       const program = new Command()
         .name('claude-sop')
         .version(readPkgVersion())
         .option('--json', 'emit stable JSON output', false)
         .exitOverride();
       registerVerbs(program);
       const jsonMode = argv.includes('--json');
       try {
         await program.parseAsync(argv);
         return ExitCode.SUCCESS;
       } catch (err) {
         if (err instanceof CommanderError) {
           if (err.code === 'commander.version' || err.code === 'commander.helpDisplayed' || err.code === 'commander.help') {
             return ExitCode.SUCCESS;
           }
           const msg = err.message || 'misuse';
           if (jsonMode) emitError(ExitCode.MISUSE, msg);
           else process.stderr.write(pc.red(msg) + '\n');
           return ExitCode.MISUSE;
         }
         if (err instanceof PreconditionError) {
           if (jsonMode) emitError(ExitCode.PRECONDITION_FAILED, err.message, (err as any).hint);
           else process.stderr.write(pc.yellow(`precondition failed: ${err.message}`) + '\n');
           return ExitCode.PRECONDITION_FAILED;
         }
         const msg = (err as Error)?.message ?? String(err);
         if (jsonMode) emitError(ExitCode.GENERIC_FAILURE, msg);
         else process.stderr.write(pc.red(`error: ${msg}`) + '\n');
         return ExitCode.GENERIC_FAILURE;
       }
     }
     ```
