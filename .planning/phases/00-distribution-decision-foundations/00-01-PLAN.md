    ```ts
    import { platform } from 'node:os';

    export function assertPlatformSupported(): void {
      const current = process.env.CLAUDE_SOP_FAKE_PLATFORM ?? platform();
      if (current === 'win32') {
        // Per INST-08 + CONTEXT.md: Windows is unsupported in v1
        process.stderr.write('claude-sop: Windows is not supported in v1. Use WSL.\n');
        process.exit(1);
      }
    }
    ```

    Create test/setup/no-network.ts using the technique from RESEARCH.md §7. It must throw a clear, identifiable error if any production code path attempts:
    - global fetch
    - node:http request
    - node:https request
    - net.Socket.prototype.connect (low-level TCP)
    - dns.lookup / dns.resolve

    Implementation: export a `installNoNetworkGuards()` function that vi.stubGlobal's fetch and monkey-patches the http/https/net/dns modules. Each stub throws `new Error('PHASE 0 NETWORK ACCESS FORBIDDEN: <which call>')`. Also export `restoreNetworkGuards()` to undo. Use vitest's `beforeAll`/`afterAll` if imported as a setup module.

    Important: this file must NOT import anything from src/. It is pure test infrastructure. Subsequent Phase 0 plans (PathResolver, Config, Scrubber) will import it via `import './setup/no-network.ts'` at the top of their test files.

        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - run: |
              node -e "const s = require('./package.json').scripts || {}; const banned = ['postinstall','preinstall','install','prepublish','prepublishOnly']; const found = banned.filter(k => k in s); if (found.length) { console.error('FORBIDDEN lifecycle scripts present:', found); process.exit(1); }"

      windows-refusal-check:
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: actions/setup-node@v4

          - run: npm install --ignore-scripts
          - run: npm run build
          - name: CLI entry must refuse win32
            run: |
              set +e
              CLAUDE_SOP_FAKE_PLATFORM=win32 node dist/cli.cjs
              code=$?
              set -e
              if [ "$code" = "0" ]; then
                echo "FAIL: CLI entry did not refuse win32 (exit code 0)"
                exit 1
              fi
              echo "OK: CLI entry refused win32 with exit code $code"
          - name: Library entry must have NO side effects (import does not exit)
